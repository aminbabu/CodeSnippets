// JavaScript Assignment Operators: =, +=, -=, *=, /=, %=, **=

// Variable declaration
var x = 6
console.log('Value of x: ' + x);


// var x += 5;
document.write('Value of x: ' + x);


// var x -= 5;
document.write('Value of x: ' + x);


// var x *= 5;
document.write('Value of x: ' + x);


// var x /= 5;
document.write('Value of x: ' + x);


// var x %= 5;
document.write('Value of x: ' + x);


// var x **= 5;
document.write('Value of x: ' + x);
